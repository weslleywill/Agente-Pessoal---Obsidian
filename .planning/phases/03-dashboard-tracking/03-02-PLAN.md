const semanaHoje = moment().format("GGGG-[W]WW");
const anoHoje = moment().format("YYYY");
const pages = dv.pages(`"01 - Diário/${anoHoje}"`)
  .where(p => p.semana === semanaHoje && p.tipo === "diario");

if (pages.length === 0) {
  dv.paragraph("_Ainda sem daily notes nesta semana. Crie a primeira com **Ctrl+P → Periodic Notes: Open today's daily note**._");
} else {
  const avg = (arr) => {
    const vals = arr.filter(v => v != null && v !== 0);
    if (vals.length === 0) return null;
    return (vals.reduce((a,b) => a+b, 0) / vals.length);
  };
  const fmt = (v) => v == null ? "_—_" : v.toFixed(1);

  const comp = avg(pages.map(p => p.pilares?.competitividade).array());
  const tal = avg(pages.map(p => p.pilares?.talento).array());
  const sau = avg(pages.map(p => p.pilares?.saude).array());
  const pre = avg(pages.map(p => p.pilares?.presenca).array());
  const pe = avg(pages.map(p => p.pilares?.performance_emocional).array());
  const todos = [comp, tal, sau, pre, pe].filter(v => v != null);
  const media = todos.length > 0 ? todos.reduce((a,b)=>a+b,0)/todos.length : null;

  dv.table(
    ["Pilar", "Média"],
    [
      ["Competitividade", fmt(comp)],
      ["Talento", fmt(tal)],
      ["Saúde", fmt(sau)],
      ["Presença", fmt(pre)],
      ["Performance Emocional", fmt(pe)],
      ["**MÉDIA GERAL**", `**${fmt(media)}**`]
    ]
  );
  dv.paragraph(`_${pages.length} daily note${pages.length > 1 ? "s" : ""} na semana._`);
}

## 📊 Log de Tempo — amostras da semana (agregado por tipo)

```dataviewjs
const semanaHoje = moment().format("GGGG-[W]WW");
const anoHoje = moment().format("YYYY");
const pages = dv.pages(`"01 - Diário/${anoHoje}"`)
  .where(p => p.semana === semanaHoje && p.tipo === "diario" && p.tarefas_log && p.tarefas_log.length > 0);

if (pages.length === 0) {
  dv.paragraph("_Nenhuma entrada em `tarefas_log` nesta semana ainda._");
  dv.paragraph("_Entradas são geradas quando você usa o comando `/loga` (implementado em Phase 4) ou edita `tarefas_log` direto no frontmatter._");
} else {
  // Achatar todas as tarefas de todos os dailies
  const todas = [];
  for (const p of pages) {
    for (const t of (p.tarefas_log ?? [])) {
      todas.push(t);
    }
  }

  // Agrupar por tipo e calcular mediana estimado vs real
  const porTipo = {};
  for (const t of todas) {
    const tipo = t.tipo ?? "outro";
    if (!porTipo[tipo]) porTipo[tipo] = [];
    porTipo[tipo].push(t);
  }

  const mediana = (arr) => {
    const limpa = arr.filter(v => v != null).sort((a,b) => a-b);
    if (limpa.length === 0) return null;
    const m = Math.floor(limpa.length / 2);
    return limpa.length % 2 === 0 ? (limpa[m-1] + limpa[m]) / 2 : limpa[m];
  };

  const linhas = [];
  for (const [tipo, arr] of Object.entries(porTipo)) {
    const est = mediana(arr.map(t => t.tempo_estimado));
    const real = mediana(arr.map(t => t.tempo_real));
    linhas.push([
      tipo,
      arr.length,
      est != null ? `${est}min` : "—",
      real != null ? `${real}min` : "—",
      (est != null && real != null) ? `${real > est ? "+" : ""}${(real - est).toFixed(0)}min` : "—"
    ]);
  }

  dv.table(
    ["Tipo", "Amostras", "Mediana estimado", "Mediana real", "Delta"],
    linhas
  );
  dv.paragraph(`_Total: ${todas.length} tarefa${todas.length === 1 ? "" : "s"} logada${todas.length === 1 ? "" : "s"} esta semana._`);
}
```

  const total = totais.reduce((a,b) => a+b, 0);
  const dias = totais.filter(v => v > 0).length;
  const diasNoMes = pages.length;
  dv.paragraph(`**Total do mês: ${total} pomodoros** em ${dias} de ${diasNoMes} dias com daily registrado.`);
  if (dias > 0) {
    dv.paragraph(`Média nos dias ativos: **${(total/dias).toFixed(1)} pomodoros/dia**`);
  }
}
